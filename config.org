* Packages

#+begin_src emacs-lisp
(require 'package)
(setq package-archives
      '(("gnu" . "https://elpa.gnu.org/packages/")
        ("melpa" . "https://melpa.org/packages/")))
(package-initialize)

(defvar my/packages
  '(evil key-chord general which-key projectile markdown-mode)
  "Packages for simple Vim-like editing and project workflow.")

(defvar my/auto-install-packages t
  "When non-nil, install missing packages automatically at startup.")

(defun my/install-missing-packages ()
  (interactive)
  (let ((missing
         (seq-filter (lambda (pkg) (not (package-installed-p pkg))) my/packages)))
    (when missing
      (unless package-archive-contents
        (package-refresh-contents))
      (dolist (pkg missing)
        (package-install pkg)))))

(let ((missing
       (seq-filter (lambda (pkg) (not (package-installed-p pkg))) my/packages)))
  (when missing
    (when my/auto-install-packages
      (my/install-missing-packages))
    (unless my/auto-install-packages
      (message "Missing packages: %s. Run M-x my/install-missing-packages"
               missing))))
#+end_src

(when (require 'evil nil t)
  (evil-mode 1)
  (setq evil-want-C-u-scroll t))

  (key-chord-mode 1)
  (setq key-chord-two-keys-delay 0.3)
  (key-chord-define evil-insert-state-map "jk" 'evil-normal-state))

(when (and (featurep 'evil) (require 'which-key nil t))
  (which-key-mode 1))

(when (and (featurep 'evil) (require 'general nil t))
  (general-auto-unbind-keys)

  (general-create-definer my/leader-def
    :keymaps 'override
    :states '(normal visual motion)
    :prefix "SPC")

  (general-create-definer my/local-leader-def
    :states '(normal visual motion)
    :prefix ",")

  (my/leader-def
    "f" '(:ignore t :which-key "files")
    "f f" #'find-file
    "f s" #'save-buffer
    "b" '(:ignore t :which-key "buffers")
    "b b" #'switch-to-buffer
    "b d" #'kill-current-buffer
    "p" '(:ignore t :which-key "projects")
    "p p" #'projectile-switch-project
    "p f" #'projectile-find-file
    "w" '(:ignore t :which-key "windows")
    "w /" #'split-window-right
    "w -" #'split-window-below
    "w d" #'delete-window
    "q" '(:ignore t :which-key "quit")
    "q q" #'save-buffers-kill-terminal
    "h" '(:ignore t :which-key "help")

      "g" #'xref-find-definitions
      "r" #'xref-find-references)))
