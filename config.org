* 说明

这是一份基于 Org-mode 文学编程的 Emacs 配置。
所有代码都在这个文件里，通过 =org-babel-tangle= 生成 =config.el=。

使用方法：
1. 把这个文件放到 =~/.emacs.d/config.org=
2. 在 =~/.emacs.d/init.el= 里写：
   #+begin_src emacs-lisp :tangle no
   (org-babel-load-file (expand-file-name "config.org" user-emacs-directory))
   #+end_src
3. 先装好依赖工具（Homebrew）：
   #+begin_src sh :tangle no
   brew install ripgrep aspell pandoc
   #+end_src
4. 重启 Emacs，一切就绪。

如果你想手动 tangle：=M-x org-babel-tangle= 或 =C-c C-v t=

* 用户设置 ← 只需要改 user-settings.el

所有个人偏好放在独立的 =~/.emacs.d/user-settings.el= 文件里。
配置主体（这个文件）完全不用动。

#+begin_src emacs-lisp
;; 加载用户设置文件，不存在则用默认值
(let ((settings (expand-file-name "user-settings.el" user-emacs-directory)))
  (if (file-exists-p settings)
      (load settings)
    (message "提示：未找到 user-settings.el，使用默认配置")))

;; 默认值（user-settings.el 里没写的项目都走这里）

;; macOS 上 Option 键作为 Meta，保留 Command 作为 Super。

** 关于 M-x 的几个技巧

- 输入 =buf sw= → 找到 =switch-to-buffer=

** Org-mode 基础配置

#+begin_src emacs-lisp
(use-package org
  :ensure nil
  :config

  ;; --- GTD 文件 ---
  (setq org-directory my/org-dir)
  (setq org-agenda-files
        (append (list my/org-inbox my/org-personal my/org-work)
                (when (bound-and-true-p my/org-extra-agenda-files)
                  my/org-extra-agenda-files)))

  ;; --- TODO 关键词 ---
  ;; 简单但够用：TODO → NEXT → DONE，WAITING，CANCELLED
  (setq org-todo-keywords
        '((sequence "TODO(t)" "NEXT(n)" "|" "DONE(d)")
          (sequence "WAITING(w@/!)" "|" "CANCELLED(c@)")))

  (setq org-todo-keyword-faces
        '(("TODO"      . (:foreground "#b45309" :weight bold))
          ("NEXT"      . (:foreground "#1d4ed8" :weight bold))
          ("WAITING"   . (:foreground "#6d28d9" :weight bold))
          ("DONE"      . (:foreground "#15803d"))
          ("CANCELLED" . (:foreground "#6b7280" :strike-through t))))

  ;; --- 标签 ---
  (setq org-tag-alist
        '((:startgroup)
          ("@home"    . ?h)
          ("@work"    . ?w)
          ("@errands" . ?e)
          ("@computer". ?c)
          (:endgroup)
          ("energy_high" . ?H)
          ("energy_low"  . ?L)
          ("quick"       . ?q)))

  ;; --- 完成时自动记录时间 ---
  (setq org-log-done 'time)
  (setq org-log-into-drawer t)

  ;; --- Refile ---
  ;; 允许 refile 到 agenda 文件的任意标题，最多 3 级
  (setq org-refile-targets
        '((org-agenda-files :maxlevel . 3)))
  (setq org-refile-use-outline-path 'file)
  (setq org-outline-path-complete-in-steps nil)  ; 一步完成，配合 vertico
  (setq org-refile-allow-creating-parent-nodes 'confirm)

  ;; --- Capture ---
  (setq org-capture-templates
        `(
          ;; 快速收集任何想法 / 任务
          ("i" "📥 Inbox" entry
           (file ,my/org-inbox)
           "* TODO %?\n:PROPERTIES:\n:CREATED: %U\n:END:\n%i\n"
           :empty-lines 1)

          ;; 个人任务
          ("p" "👤 Personal 任务" entry
           (file+headline ,my/org-personal "Tasks")
           "* TODO %?\n:PROPERTIES:\n:CREATED: %U\n:END:\n"
           :empty-lines 1)

          ;; 工作任务
          ("w" "💼 Work 任务" entry
           (file+headline ,my/org-work "Tasks")
           "* TODO %?\n:PROPERTIES:\n:CREATED: %U\n:END:\n"
           :empty-lines 1)

          ;; 快速笔记（不是任务）
          ("n" "📝 笔记" entry
           (file+headline ,my/org-inbox "Notes")
           "* %? :note:\n%U\n%i\n"
           :empty-lines 1)

          ;; 日记 / 日志
          ("j" "📓 日记" entry
           (file+datetree ,my/org-journal)
           "* %U\n%?\n"
           :empty-lines 1)

          ;; 读书 / 阅读笔记
          ("r" "📚 阅读笔记" entry
           (file+headline ,my/org-personal "Reading")
           "* %^{书名 / 文章}\n:PROPERTIES:\n:CREATED: %U\n:SOURCE: %^{来源}\n:END:\n%?\n"
           :empty-lines 1)))

  ;; --- 外观 ---
  (setq org-hide-emphasis-markers t    ; 隐藏 *bold* 的星号
        org-pretty-entities t          ; 显示 LaTeX 符号
        org-ellipsis " ▾"
        org-startup-indented t         ; 缩进对齐
        org-src-fontify-natively t     ; 代码块语法高亮
        org-src-tab-acts-natively t
        org-confirm-babel-evaluate nil)

  )
#+end_src

** Agenda 视图 — 你的每日仪表盘

#+begin_src emacs-lisp
(with-eval-after-load 'org-agenda
  (setq org-agenda-custom-commands
        '(
          ;; "d" → 今日仪表盘（最常用）
          ("d" "📅 今日仪表盘"
           ((agenda ""
                    ((org-agenda-span 1)
                     (org-agenda-show-all-dates nil)
                     (org-agenda-start-on-weekday nil)))
            (todo "NEXT"
                  ((org-agenda-overriding-header "⚡ 下一步行动")
                   (org-agenda-skip-function
                    '(org-agenda-skip-entry-if 'scheduled 'deadline))))
            (todo "WAITING"
                  ((org-agenda-overriding-header "⏳ 等待中")))
            (tags "inbox"
                  ((org-agenda-overriding-header "📥 待处理 Inbox")
                   (org-tags-match-list-sublevels nil)))))

          ;; "w" → 本周视图
          ("W" "📆 本周概览"
           ((agenda ""
                    ((org-agenda-span 7)
                     (org-agenda-start-on-weekday 1)))))

          ;; "p" → 个人项目
          ("p" "👤 个人任务"
           ((tags-todo "@home|@errands|@computer"
                       ((org-agenda-overriding-header "个人待办")))))

          ;; "k" → 工作
          ("k" "💼 工作任务"
           ((tags-todo "@work"
                       ((org-agenda-overriding-header "工作待办")))))))

  ;; Agenda 显示设置
  (setq org-agenda-window-setup 'current-window
        org-agenda-block-separator ?─
        org-agenda-time-grid '((daily today require-timed)
                               (800 1000 1200 1400 1600 1800 2000)
                               " ┄┄┄┄┄ " "┄┄┄┄┄┄┄┄┄┄┄┄┄┄┄")
        org-agenda-current-time-string "◀── 现在"
        org-agenda-prefix-format
        '((agenda . " %i %-12:c%?-12t% s")
          (todo   . " %i %-12:c")
          (tags   . " %i %-12:c")
          (search . " %i %-12:c"))))

* 剪贴板 — 与 macOS 系统互通

macOS GUI 版 Emacs 通常能自动同步剪贴板，但终端版（=emacs -nw=）或某些情况下会断开。这节把两种情况都覆盖好。

#+begin_src emacs-lisp
;; GUI 版：直接用系统剪贴板（通常默认已开，显式写出更保险）
(setq select-enable-clipboard t
      select-enable-primary   nil)   ; macOS 没有 X11 primary selection，关掉

;; 终端版（emacs -nw）：用 pbcopy / pbpaste 桥接系统剪贴板
(unless (display-graphic-p)
  (defun my/pbcopy ()
    "把 Emacs kill-ring 最新内容复制到 macOS 系统剪贴板。"
    (interactive)
    (let ((text (current-kill 0 t)))
      (with-temp-buffer
        (insert text)
        (call-process-region (point-min) (point-max) "pbcopy"))))

  (defun my/pbpaste ()
    "把 macOS 系统剪贴板内容粘贴到 Emacs，并推入 kill-ring。"
    (interactive)
    (let ((text (shell-command-to-string "pbpaste")))
      (kill-new text)
      (insert text)))

  ;; 覆盖终端模式下的复制/粘贴行为
  (global-set-key (kbd "M-w") (lambda ()
                                (interactive)
                                (call-interactively 'kill-ring-save)
                                (my/pbcopy)))
  (global-set-key (kbd "C-w") (lambda ()
                                (interactive)
                                (call-interactively 'kill-region)
                                (my/pbcopy)))
  (global-set-key (kbd "C-y") 'my/pbpaste))

;; consult-yank-pop：Cmd+Y 弹出剪贴板历史，可以粘贴任意一条历史记录
;; （已在命令搜索一节绑定到 M-y，这里作为提示保留）
#+end_src

** 日常操作说明

| 操作 | GUI Emacs | 终端 Emacs |
|------|-----------|------------|
| 复制到系统剪贴板 | =M-w= | =M-w=（自动调 pbcopy）|
| 从系统剪贴板粘贴 | =C-y= | =C-y=（自动调 pbpaste）|
| 粘贴历史记录 | =M-y= 弹出列表 | =M-y= 弹出列表 |

=M-y=（即 =Cmd+Y=）弹出的是 =consult-yank-pop=，可以看到你最近复制过的所有内容，用方向键选择后回车粘贴，非常好用。
