* 说明

这是一份基于 Org-mode 文学编程的 Emacs 配置。
所有代码都在这个文件里，通过 =org-babel-tangle= 生成 =config.el=。

使用方法：
1. 把这个文件放到 =~/.emacs.d/config.org=
2. 在 =~/.emacs.d/init.el= 里写：
   #+begin_src emacs-lisp :tangle no
   (org-babel-load-file (expand-file-name "config.org" user-emacs-directory))
   #+end_src
3. 先装好依赖工具（Homebrew）：
   #+begin_src sh :tangle no
   brew install ripgrep aspell pandoc
   #+end_src
4. 重启 Emacs，一切就绪。

如果你想手动 tangle：=M-x org-babel-tangle= 或 =C-c C-v t=

;; macOS 上 Option 键作为 Meta，保留 Command 作为 Super。
;; 这样 Option+x 是 M-x，Command 不再和 Meta 快捷键冲突。
(setq mac-command-modifier 'super
      mac-option-modifier  'meta

** 关于 M-x 的几个技巧

=M-x= 现在可以这样搜索：

- 输入 =buf sw= → 找到 =switch-to-buffer=
- 输入 =gtd= → 找到你自定义的 GTD 命令（如果有前缀）

** Org-mode 基础配置

#+begin_src emacs-lisp
(use-package org
  :ensure nil
  :config

  ;; --- GTD 文件 ---
  (setq org-directory my/org-dir)
  (setq org-agenda-files
        (append (list my/org-inbox my/org-tasks)
                (when (bound-and-true-p my/org-extra-agenda-files)
                  my/org-extra-agenda-files)))

  ;; --- TODO 关键词 ---
  ;; Things/GTD 风格：Inbox 收集 TODO，整理后进入 NEXT/WAITING/SOMEDAY。
  (setq org-todo-keywords
        '((sequence "TODO(t)" "NEXT(n)" "|" "DONE(d)")
          (sequence "WAITING(w@/!)" "SOMEDAY(s)" "|" "CANCELLED(c@)")))

  (setq org-todo-keyword-faces
        '(("TODO"      . (:foreground "#b45309" :weight bold))
          ("NEXT"      . (:foreground "#1d4ed8" :weight bold))
          ("WAITING"   . (:foreground "#6d28d9" :weight bold))
          ("SOMEDAY"   . (:foreground "#64748b" :weight bold))
          ("DONE"      . (:foreground "#15803d"))
          ("CANCELLED" . (:foreground "#6b7280" :strike-through t))))

  ;; --- 标签 ---
  (setq org-tag-alist
        '(("personal" . ?p)
          ("work"     . ?w)
          ("home"     . ?h)
          ("errands"  . ?e)
          ("computer" . ?c)
          ("health"   . ?m)
          ("learning" . ?l)
          ("travel"   . ?v)
          ("energy_high" . ?H)
          ("energy_low"  . ?L)
          ("quick"       . ?q)))

  ;; --- 完成时自动记录时间 ---
  (setq org-log-done 'time)
  (setq org-log-into-drawer t)

  ;; --- Archive ---
  (setq org-archive-location (concat my/org-dir "/archive.org::* Archived"))

  ;; --- Refile ---
  ;; 只允许 refile 到各文件的一级容器标题（Inbox / Tasks / Ideas / Reading）。
  ;; 任务本身从二级开始，避免误把一个任务 refile 到另一个任务下面。
  (setq org-refile-targets
        '((org-agenda-files :level . 1)))
  (setq org-refile-use-outline-path t)
  (setq org-outline-path-complete-in-steps nil)  ; 一步完成，配合 vertico
  (setq org-refile-allow-creating-parent-nodes 'confirm)

  ;; --- Capture ---
  (setq org-capture-templates
        `(
          ;; 快速收集任何想法 / 任务
          ("i" "📥 Inbox" entry
           (file+headline ,my/org-inbox "Inbox")
           "* TODO %?\n:PROPERTIES:\n:CREATED: %U\n:END:\n%i\n"
           :empty-lines 1)

          ;; 个人任务
          ("p" "👤 Personal 任务" entry
           (file+headline ,my/org-tasks "Tasks")
           "* TODO %? :personal:\n:PROPERTIES:\n:CREATED: %U\n:END:\n"
           :empty-lines 1)

          ;; 工作任务
          ("w" "💼 Work 任务" entry
           (file+headline ,my/org-tasks "Tasks")
           "* TODO %? :work:\n:PROPERTIES:\n:CREATED: %U\n:END:\n"
           :empty-lines 1)

          ;; Someday / Maybe
          ("s" "🕰 Someday" entry
           (file+headline ,my/org-tasks "Tasks")
           "* SOMEDAY %?\n:PROPERTIES:\n:CREATED: %U\n:END:\n"
           :empty-lines 1)

          ;; 碎碎念 / 想法（不是任务）
          ("n" "💭 Ideas / 碎碎念" entry
           (file+headline ,my/org-ideas "Ideas")
           "* %?\n:PROPERTIES:\n:CREATED: %U\n:END:\n%i\n"
           :empty-lines 1)

          ;; 读书 / 阅读笔记
          ("r" "📚 阅读笔记" entry
           (file+headline ,my/org-ideas "Reading")
           "* %^{书名 / 文章}\n:PROPERTIES:\n:CREATED: %U\n:SOURCE: %^{来源}\n:END:\n%?\n"
           :empty-lines 1)))

  ;; --- 外观 ---
  (setq org-hide-emphasis-markers t    ; 隐藏 *bold* 的星号
        org-pretty-entities t          ; 显示 LaTeX 符号
        org-ellipsis " ▾"
        org-startup-indented t         ; 缩进对齐
        org-src-fontify-natively t     ; 代码块语法高亮
        org-src-tab-acts-natively t
        org-confirm-babel-evaluate t)

  )
#+end_src

** 初始化 org 文件目录

#+begin_src emacs-lisp
(defun my/gtd-file-title (path)
  "Return a readable Org title for PATH."
  (file-name-sans-extension (file-name-nondirectory path)))

(defun my/gtd-create-file (path &optional body)
  "Create PATH with a standard Org header and optional BODY."
  (unless (file-exists-p path)
    (with-temp-file path
      (insert (format "#+TITLE: %s\n#+STARTUP: overview\n\n%s"
                      (my/gtd-file-title path)
                      (or body ""))))))

(defun my/gtd-ensure-top-level-headings (path headings)
  "Append missing top-level HEADINGS to PATH."
  (let ((buf (find-file-noselect path)))
    (with-current-buffer buf
      (save-excursion
        (dolist (heading headings)
          (unless (save-excursion
                    (goto-char (point-min))
                    (re-search-forward
                     (format "^\\* %s\\(?:[ \t\n]\\|$\\)" (regexp-quote heading))
                     nil t))
            (goto-char (point-max))
            (unless (bolp) (insert "\n"))
            (insert (format "\n* %s\n" heading)))))
      (when (buffer-modified-p)
        (save-buffer)))
    (kill-buffer buf)))

(defun my/gtd-initialize ()
  "创建 GTD 所需的 org 文件（如果不存在）。"
  (interactive)
  (make-directory my/org-dir t)
  (dolist (path (list my/org-inbox my/org-ideas))
    (my/gtd-create-file path))
  (my/gtd-ensure-top-level-headings my/org-inbox '("Inbox"))
  (my/gtd-ensure-top-level-headings my/org-ideas '("Ideas"))

* 剪贴板 — 与 macOS 系统互通

macOS GUI 版 Emacs 通常能自动同步剪贴板，但终端版（=emacs -nw=）或某些情况下会断开。这节把两种情况都覆盖好。

#+begin_src emacs-lisp
;; GUI 版：直接用系统剪贴板（通常默认已开，显式写出更保险）
(setq select-enable-clipboard t
      select-enable-primary   nil)   ; macOS 没有 X11 primary selection，关掉

;; 终端版（emacs -nw）：用 pbcopy / pbpaste 桥接系统剪贴板
(unless (display-graphic-p)
  (defun my/pbcopy ()
    "把 Emacs kill-ring 最新内容复制到 macOS 系统剪贴板。"
    (interactive)
    (let ((text (current-kill 0 t)))
      (with-temp-buffer
        (insert text)
        (call-process-region (point-min) (point-max) "pbcopy"))))

  (defun my/pbpaste ()
    "把 macOS 系统剪贴板内容粘贴到 Emacs，并推入 kill-ring。"
    (interactive)
    (let ((text (shell-command-to-string "pbpaste")))
      (kill-new text)
      (insert text)))

  ;; 覆盖终端模式下的复制/粘贴行为
  (global-set-key (kbd "M-w") (lambda ()
                                (interactive)
                                (call-interactively 'kill-ring-save)
                                (my/pbcopy)))
  (global-set-key (kbd "C-w") (lambda ()
                                (interactive)
                                (call-interactively 'kill-region)
                                (my/pbcopy)))
  (global-set-key (kbd "C-y") 'my/pbpaste))

;; （已在命令搜索一节绑定到 M-y，这里作为提示保留）
#+end_src

| 操作 | GUI Emacs | 终端 Emacs |
|------|-----------|------------|
| 复制到系统剪贴板 | =M-w= | =M-w=（自动调 pbcopy）|

* capture-llm — 自然语言智能捕获

用 LLM 自动分类自然语言输入，写入对应的 org 文件。

#+begin_src emacs-lisp
;; 确保 quelpa 已安装
(unless (package-installed-p 'quelpa)
  (package-install 'quelpa))

;; 从 GitHub 安装 capture-llm（新装机自动拉取，无需手动 clone）
(unless (package-installed-p 'capture-llm)
  (quelpa '(capture-llm :fetcher github :repo "danliustc/capture-llm")))

(require 'capture-llm)
(require 'llm-deepseek)   ; make-llm-deepseek 不会被 autoload
(setq llm-warn-on-nonfree nil)         ; 关闭非自由 API 警告

;; DeepSeek provider（API key 在 user-settings.el 里设置 my/deepseek-api-key）
(when (bound-and-true-p my/deepseek-api-key)
  (setq capture-llm-provider
        (make-llm-deepseek :key my/deepseek-api-key
                           :chat-model "deepseek-chat")))

;; 快捷键：C-c C-l
(global-set-key (kbd "C-c C-l") 'capture-llm-capture)
#+end_src

使用方法：按 =C-c C-l=，输入一句话（如"明天下午3点去医院体检"），LLM 自动判断分类、打标签、提取时间，预览后确认写入。

需要在 =user-settings.el= 里设置 =my/deepseek-api-key=。
