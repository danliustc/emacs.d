* Packages

#+begin_src emacs-lisp
(require 'package)
(setq package-archives
      '(("gnu" . "https://elpa.gnu.org/packages/")
        ("melpa" . "https://melpa.org/packages/")))
(package-initialize)

(defvar my/packages
  '(evil key-chord general which-key projectile markdown-mode ayu-theme)
  "Packages for simple Vim-like editing and project workflow.")

(defvar my/auto-install-packages t
  "When non-nil, install missing packages automatically at startup.")

(defun my/install-missing-packages ()
  (interactive)
  (let ((missing
         (seq-filter (lambda (pkg) (not (package-installed-p pkg))) my/packages)))
    (when missing
      (unless package-archive-contents
        (package-refresh-contents))
      (dolist (pkg missing)
        (package-install pkg)))))

(let ((missing
       (seq-filter (lambda (pkg) (not (package-installed-p pkg))) my/packages)))
  (when missing
    (when my/auto-install-packages
      (my/install-missing-packages))
    (unless my/auto-install-packages
      (message "Missing packages: %s. Run M-x my/install-missing-packages"
               missing))))
#+end_src

* Project / Workspace

#+begin_src emacs-lisp
(if (require 'projectile nil t)
    (progn
      (defun my/projectile-find-recent-file-in-project (project-root)
        (let* ((default-directory project-root)
               (recent-files (seq-take (projectile-recently-active-files) 8)))
          (if recent-files
              (let ((selected-file
                     (minibuffer-with-setup-hook
                         (lambda ()
                           (when (active-minibuffer-window)
                             (minibuffer-completion-help)))
                       (completing-read "Recent file in project: "
                                        recent-files nil t nil nil
                                        (car recent-files)))))
                (when (and selected-file (not (string= selected-file "")))
                  (find-file (expand-file-name selected-file project-root))))
            (let ((default-directory project-root))
              (call-interactively #'projectile-find-file)))))

      (defun my/projectile-switch-project-recent ()
        (interactive)
        (let* ((known-projects (projectile-known-projects))
               (current-project (ignore-errors
                                  (abbreviate-file-name (projectile-project-root))))
               (ordered-projects
                (if (and current-project (member current-project known-projects))
                    (cons current-project (delete current-project known-projects))
                  known-projects))
               (recent-projects (seq-take ordered-projects 5)))
          (if recent-projects
              (let ((selected-project
                     (minibuffer-with-setup-hook
                         (lambda ()
                           (when (active-minibuffer-window)
                             (minibuffer-completion-help)))
                       (completing-read "Switch to recent project: "
                                        recent-projects nil t nil nil
                                        (car recent-projects)))))
                (when (and selected-project (not (string= selected-project "")))
                  (let ((project-root (expand-file-name selected-project)))
                    (let ((projectile-switch-project-action #'ignore))
                      (projectile-switch-project-by-name project-root))
                    (my/projectile-find-recent-file-in-project project-root))))
            (call-interactively #'projectile-switch-project))))
      (projectile-mode 1)
      (setq projectile-completion-system 'default)
      (define-key global-map (kbd "C-c p p") #'my/projectile-switch-project-recent)
      (define-key global-map (kbd "C-c p f") #'projectile-find-file))
  (progn
    (define-key global-map (kbd "C-c p p") #'project-switch-project)
    (define-key global-map (kbd "C-c p f") #'project-find-file)))
#+end_src
