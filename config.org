#+begin_src emacs-lisp
(require 'package)
(setq package-archives
      '(("gnu" . "https://elpa.gnu.org/packages/")
        ("melpa" . "https://melpa.org/packages/")))
(package-initialize)

(defvar my/packages
  '(evil key-chord general which-key projectile markdown-mode ayu-theme
         vertico orderless consult marginalia embark embark-consult)
  "Packages for simple Vim-like editing and project workflow.")

(defvar my/auto-install-packages t
  "When non-nil, install missing packages automatically at startup.")

(defun my/install-missing-packages ()
  (interactive)
  (let ((missing
         (seq-filter (lambda (pkg) (not (package-installed-p pkg))) my/packages)))
    (when missing
      (condition-case err
          (progn
            (unless package-archive-contents
              (package-refresh-contents))
            (dolist (pkg missing)
              (condition-case pkg-err
                  (package-install pkg)
                (error
                 (display-warning
                  'my/init
                  (format "Failed to install %s: %s"
                          pkg (error-message-string pkg-err))
                  :warning)))))
        (error
         (display-warning
          'my/init
          (format "Package bootstrap failed: %s" (error-message-string err))
          :warning))))))

(let ((missing
       (seq-filter (lambda (pkg) (not (package-installed-p pkg))) my/packages)))
  (when missing
    (if my/auto-install-packages
        (my/install-missing-packages)
      (message "Missing packages: %s. Run M-x my/install-missing-packages"
               missing))))

#+begin_src emacs-lisp
(defconst my/orgfiles-default-root
  (expand-file-name "orgfiles/" "~/Documents/")
  "Portable fallback directory for personal Org notes.")

(defvar my/orgfiles-root nil
  "Root directory for personal Org notes.
May be pre-set via `local.el' or `ORGFILES_ROOT'; normalized below.")

(setq my/orgfiles-root
      (file-name-as-directory
       (expand-file-name
        (or (getenv "ORGFILES_ROOT")
            my/orgfiles-root
            my/orgfiles-default-root))))

(defvar my/orgfiles-idea-file (expand-file-name "ideas.org" my/orgfiles-root)
  "Single file for daily idea capture.")

(defvar my/orgfiles-inbox-file (expand-file-name "inbox.org" my/orgfiles-root)
  "Single file for quick TODO capture.")

(defvar my/orgfiles-projects-dir
  (file-name-as-directory (expand-file-name "projects" my/orgfiles-root))
  "Directory of active project Org files scanned by the agenda.
Meeting, brainstorm and journal files in the root are intentionally
excluded so the agenda stays small; refile important items here.")

(defun my/orgfiles--ensure-dir (dir)
  (unless (file-directory-p dir)
    (make-directory dir t)))

(defun my/orgfiles--safe-file-stem (name)
  (let* ((raw-name (string-trim name))
         (base-name (file-name-nondirectory raw-name))
         (name-without-ext (file-name-sans-extension base-name))
         (space-normalized (replace-regexp-in-string "[[:space:]]+" "-" name-without-ext))
         (safe-name
          (replace-regexp-in-string
           "[^[:alnum:]_[:multibyte:]-]"
           "-"
           space-normalized))
         (collapsed (replace-regexp-in-string "-+" "-" safe-name))
         (trimmed (string-trim collapsed "-+" "-+")))
    (if (string-empty-p trimmed) "untitled" trimmed)))

(defun my/orgfiles--timestamped-file-path (dir title &optional time)
  (my/orgfiles--ensure-dir dir)
  (let* ((captured-at (or time (current-time)))
         (timestamp (format-time-string "%Y-%m-%d-%H%M" captured-at))
         (slug (my/orgfiles--safe-file-stem title))
         (counter 0)
         candidate)
    (while
        (progn
          (setq candidate
                (expand-file-name
                 (if (= counter 0)
                     (format "%s-%s.org" timestamp slug)
                   (format "%s-%s-%d.org" timestamp slug counter))
                 dir))
          (setq counter (1+ counter))
          (file-exists-p candidate)))
    candidate))

(defun my/orgfiles--meeting-template (meeting-name meeting-time)
  (let ((captured-at (or meeting-time (current-time)))
        (deadline (format-time-string "%Y-%m-%d %a"
                                      (time-add (or meeting-time (current-time))
                                                (days-to-time 7)))))
    (format "#+TITLE: %s\n#+DATE: <%s>\n\n* 背景\n\n* 结论（Decision）\n\n* 待办（Action Items）\n** TODO [#A] xxx\nOWNER: 负责人\nDEADLINE: <%s>\n\n* 风险/阻塞\n\n* 下次会议前要准备\n"
            meeting-name
            (format-time-string "%Y-%m-%d %a %H:%M" captured-at)
            deadline)))

(defun my/orgfiles--today-journal-file ()
  (expand-file-name (format-time-string "%Y-%m-%d.org") my/orgfiles-root))

(defun my/orgfiles--new-brainstorm-file (title)
  (let* ((display-title (let ((trimmed (string-trim title)))
                          (if (string-empty-p trimmed) "Untitled" trimmed)))
         (candidate (my/orgfiles--timestamped-file-path my/orgfiles-root title)))
    (with-temp-file candidate
      (insert (format "#+TITLE: Brainstorm - %s\n#+DATE: <%s>\n\n* Context\n\n* Ideas\n\n* Next Steps\n"
                      display-title
                      (format-time-string "%Y-%m-%d %a"))))
    candidate))

(defun my/orgfiles--new-meeting-file (meeting-name &optional meeting-time)
  (let* ((captured-at (or meeting-time (current-time)))
         (target-file (my/orgfiles--timestamped-file-path my/orgfiles-root
                                                          meeting-name
                                                          captured-at)))
    (with-temp-file target-file
      (insert (my/orgfiles--meeting-template meeting-name captured-at)))
    target-file))

(defun my/orgfiles--ensure-journal-file ()
  (let ((journal-file (my/orgfiles--today-journal-file)))
    (my/orgfiles--ensure-dir my/orgfiles-root)
    (unless (file-exists-p journal-file)
      (with-temp-file journal-file
        (insert (format "#+TITLE: Journal %s\n#+DATE: <%s>\n\n* Journal\n"
                        (format-time-string "%Y-%m-%d")
                        (format-time-string "%Y-%m-%d %a")))))
    journal-file))

(defun my/orgfiles-capture-idea ()
  (interactive)
  (require 'org-capture)
  (my/orgfiles--ensure-dir my/orgfiles-root)
  (unless (file-exists-p my/orgfiles-idea-file)
    (with-temp-file my/orgfiles-idea-file
      (insert "#+TITLE: Ideas\n\n* Inbox\n")))
  (org-capture nil "i"))

(defun my/orgfiles-capture-todo ()
  (interactive)
  (require 'org-capture)
  (my/orgfiles--ensure-dir my/orgfiles-root)
  (unless (file-exists-p my/orgfiles-inbox-file)
    (with-temp-file my/orgfiles-inbox-file
      (insert "#+TITLE: Inbox\n\n* Tasks\n")))
  (org-capture nil "t"))

(defun my/orgfiles-capture-brainstorm ()
  (interactive)
  (let ((title (read-string "Brainstorm title: ")))
    (find-file (my/orgfiles--new-brainstorm-file title))))

(defun my/orgfiles-capture-journal ()
  (interactive)
  (require 'org-capture)
  (my/orgfiles--ensure-journal-file)
  (org-capture nil "j"))

(defun my/orgfiles-capture-meeting ()
  (interactive)
  (require 'org)
  (let* ((meeting-name (read-string "Meeting name: "))
         (meeting-time (org-read-date nil t nil "Meeting time: " (current-time))))
    (find-file (my/orgfiles--new-meeting-file meeting-name meeting-time))))

(defun my/orgfiles-capture-dispatch ()
  (interactive)
  (let ((capture-type (completing-read "Capture type: "
                                       '("todo" "brainstorm" "idea" "journal" "meeting")
                                       nil t nil nil "todo")))
    (pcase capture-type
      ("todo" (my/orgfiles-capture-todo))
      ("brainstorm" (my/orgfiles-capture-brainstorm))
      ("idea" (my/orgfiles-capture-idea))
      ("journal" (my/orgfiles-capture-journal))
      ("meeting" (my/orgfiles-capture-meeting)))))

(with-eval-after-load 'org
  (my/orgfiles--ensure-dir my/orgfiles-projects-dir)
  (setq org-directory my/orgfiles-root
        org-agenda-files (list my/orgfiles-inbox-file
                               my/orgfiles-idea-file
                               my/orgfiles-projects-dir)
        org-default-notes-file my/orgfiles-idea-file
        org-refile-targets '((org-agenda-files :maxlevel . 3))
        org-refile-use-outline-path 'file
        org-outline-path-complete-in-steps nil
        org-refile-allow-creating-parent-nodes 'confirm
        org-capture-templates
        '(("t" "Todo" entry
           (file+headline my/orgfiles-inbox-file "Tasks")
           "* TODO %?\n%U\n")
          ("i" "Idea" entry
           (file+headline my/orgfiles-idea-file "Inbox")
           "* %U %?\n")
          ("j" "Journal" entry
           (file+headline (lambda () (my/orgfiles--ensure-journal-file)) "Journal")
           "* %U %?\n"))))
